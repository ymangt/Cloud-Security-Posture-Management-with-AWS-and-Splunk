#!/bin/bash
# prowler-install.sh - Script to install Prowler on Amazon Linux 2023

# Update system and install dependencies
sudo dnf update -y
sudo dnf install -y git python3 python3-pip

# Install Prowler
git clone https://github.com/prowler-cloud/prowler.git
cd prowler
sudo pip3 install -r requirements.txt
sudo ./install-aws.sh

# Verify installation
./prowler-cli.py --version
